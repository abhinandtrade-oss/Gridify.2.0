Google Drive Integration

Follow these steps to enable file saving to Google Drive.

How to setup:
  1. Copy the script provided below.
  2. Go to Google Apps Script.
  3. Create a new project and paste the script.
  4. Click Deploy > New Deployment.
  5. Select type Web App.
  6. Set "Who has access" to Anyone.
  7. Copy the Web App URL and paste it in the field below.
// Google Apps Script - SSV News Upload Bridge
const DRAFT_FOLDER_ID = '1_GKHlr20kAN340Dmf4Hhqk7sGnP_4D0J';
const PUBLISHED_FOLDER_ID = '165DnJEi9jCcd-FirUpIgdCkzHKfY36w3';

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const draftFolder = DriveApp.getFolderById(DRAFT_FOLDER_ID);
    const publishedFolder = DriveApp.getFolderById(PUBLISHED_FOLDER_ID);
    
    // Handle Delete Actions
    if (data.action === 'delete' || data.action === 'deleteDrafts') {
      const searchPattern = data.pattern;
      
      // Cleaner helper
      const cleanFolder = (folder) => {
        const files = folder.getFiles();
        let count = 0;
        while (files.hasNext()) {
          const file = files.next();
          if (file.getName().includes(searchPattern)) {
             file.setTrashed(true);
             count++;
          }
        }
        return count;
      };

      let removed = cleanFolder(draftFolder);
      if (data.action === 'delete') {
        removed += cleanFolder(publishedFolder);
      }
      
      return ContentService.createTextOutput(JSON.stringify({
        success: true,
        message: `Removed ${removed} files for pattern ${searchPattern}`
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
    // Default: Handle Upload Action
    const targetFolder = data.status === 'published' ? publishedFolder : draftFolder;
    
    const contentType = data.mimetype;
    const bytes = Utilities.base64Decode(data.base64);
    const blob = Utilities.newBlob(bytes, contentType, data.filename);
    
    const file = targetFolder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    
    const fileUrl = `https://drive.google.com/thumbnail?id=${file.getId()}&sz=w1000`;
    
    return ContentService.createTextOutput(JSON.stringify({
      success: true,
      fileId: file.getId(),
      url: fileUrl
    })).setMimeType(ContentService.MimeType.JSON);
    
  } catch (f) {
    return ContentService.createTextOutput(JSON.stringify({
      success: false,
      error: f.message
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

function doOptions(e) {
  return ContentService.createTextOutput("")
    .setMimeType(ContentService.MimeType.TEXT);
}
